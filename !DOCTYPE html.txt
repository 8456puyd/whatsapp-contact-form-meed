<script>
    function sendMessage() {
        var fullName = document.getElementById('fullName').value;
        var phoneNumber = document.getElementById('phoneNumber').value;
        var serviceType = document.getElementById('serviceType').value;
        var marketType = document.getElementById('marketType').value;

        if (fullName && phoneNumber && serviceType && marketType) {
            var whatsappNumber = '212691804398'; // رقم واتساب الخاص بك
            
            var message = "الاسم: " + fullName + "%0A" +
                          "رقم الهاتف: " + phoneNumber + "%0A" +
                          "نوع الخدمة: " + serviceType + "%0A" +
                          "السوق: " + marketType;
                          
            var whatsappLink = "https://wa.me/" + whatsappNumber + "?text=" + message;
            
            // فتح رابط واتساب
            window.open(whatsappLink, '_blank');
        } else {
            alert('يرجى ملء جميع الحقول واختيار الخدمة والسوق.');
        }
    }
</script>
